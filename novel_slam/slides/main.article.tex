%! Tex Root=./main.article.tex
\documentclass[
	a4paper,10pt,
	twoside,openright
]{memoir}
\DisemulatePackage{enumitem} % for beamer compatability

\usepackage{geometry}
\geometry{
	total={210mm,297mm},
	left=20mm,right=20mm,
	bindingoffset=10mm, top=25mm, bottom=25mm
}

\maxtocdepth{subsubsection} % the depth of tableofcontents, default `section'.
\setsecnumdepth{subsubsection} % the depth of auto-index of sections, default `section'.

\usepackage[noxcolor]{beamerarticle}
\setjobnamebeamerversion{main.beamer}

\usepackage[dvipsnames,pst]{xcolor}
\usepackage{tikz}
\usepackage{tcolorbox}

% color palette
\definecolor{um-blue}{HTML}{002855}
\definecolor{um-gold}{HTML}{84754E}
\definecolor{um-red}{HTML}{Ef3340}
\definecolor{um-yellow}{HTML}{84754E}

\colorlet{color-frametitle}{um-blue}
\colorlet{color-progressbar}{um-gold}

% frametitle as subsection
\newcommand*\Frametitle[1]{\subsection{#1}}

\mode<article>{
	\setbeamertemplate{block begin}
	{
		\begin{tcolorbox}
			[%
				colback=lightgray!3,
				colframe=color-frametitle,
				arc=1.5mm,
				title=\insertblocktitle,
				colbacktitle=white,
				coltitle=black,
				fonttitle={\large\bfseries\color{color-progressbar}},
				detach title,
				before upper={\tcbtitle\smallskip\par}
			]
			}
			\setbeamertemplate{block end}{
		\end{tcolorbox}
	}
}

% footnote style for memoir
\footmarkstyle{%
  \makebox[\parindent][l]{\textsuperscript{#1}}%
}
\setlength{\footmarkwidth}{0pt}
\setlength{\footmarksep}  {0pt}
\setlength{\footparindent}{\parindent}

% footnote without counter
\newcommand\blfootnote[1]{%
	\begingroup
	\renewcommand\thefootnote{}\footnote{#1}%
	\addtocounter{footnote}{-1}%
	\endgroup
}
